# How to logout i3wm

	M-S-e

# Open new terminal

	M-Enter
	
# Swith to left/right terminal

	M-j
	M-;
	
# Swith to up/down terminal

	M-l
	M-k

# Split Vertically/Horizontally

	M-v
	M-h

# Swith mode

	M-e 

# Stacking

	M-s 

# Tabbed

	M-w 
	
# Full Screen	

	M-f 

# dmenu

	M-d

# Close windows

	M-S-q 

	C-w

# Workspace 

	M-1,2,3... 

# Move window to another workspace

	M-S-1,2,3...

# Restart i3

	M-S-r
	
# Floating/Exit Floating

	M-S-space
